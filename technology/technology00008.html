    js样例

    $(function(){
               $("#cargotestXL").bsSuggest('init', {
                      url: prefix + "/cargoinfo",    //请求数据的 URL 地址
                      effectiveFields: ["cargoId", "cargoName"],    // 有效显示于列表中的字段，非有效字段都会过滤，默认全部有效。
                      searchFields: ["cargoId", "cargoName"],    // 是否双向匹配搜索。为 true 即输入关键字包含或包含于匹配字段均认为匹配成功，为 false 则输入关键字包含于匹配字段认为匹配成功
                      effectiveFieldsAlias:{cargoId: "编号",cargoName: "名称"},    // 有效字段的别名对象，用于 header 的显示
                      clearable: true,    // 是否可清除已输入的内容
                        hideOnSelect: true, 
                      idField: "cargoId",    //每组数据的哪个字段作为 data-id，优先级高于 indexId 设置（推荐）
                      keyField: "cargoName"    //每组数据的哪个字段作为输入框内容，优先级高于 indexKey 设置（推荐）
                  }).on('onDataRequestSuccess', function (e, result) {    //当 AJAX 请求数据成功时触发，并传回结果到第二个参数
                      console.log('onDataRequestSuccess: ', result);
                  }).on('onSetSelectValue', function (e, selectedData, selectedRawData) {    //当从下拉菜单选取值时触发，并传回设置的数据到第二个参数
                      //拿到下拉框id,给到隐藏域
                      $("#cargoId").val(selectedData.id);
                      //放到隐藏域   后台改一下
                      console.log('onSetSelectValue: ', e.target.value, selectedData, selectedRawData);
                  }).on('onUnsetSelectValue', function () {    //当设置了 idField，且自由输入内容时触发（与背景警告色显示同步）
                      $("#cargoId").val("");
                      console.log('onUnsetSelectValue');
                  }).on('onShowDropdown', function (e, data) {    //下拉菜单显示时触发
                      console.log('onShowDropdown', e.target.value, data);
                  }).on('onHideDropdown', function (e, data) {    //下拉菜单隐藏式触发
                      console.log('onHideDropdown', e.target.value, data);
                  });

    6. indexId: 0, //每组数据的第几个数据，作为input输入框的 data-id，设为 -1 且 idField 为空则不设置此值

    url 获取数据时，若格式不符合则要对数据的处理

    processData: function(json) { // url 获取数据时，对数据的处理，作为 getData 的回调函数 
        var i, len, data = { value: [] };
        if(!json || json.length == 0) {
            return false;
        }
        len = json.length;
        for(i = 0; i < len; i++) {
            data.value.push({
                "name": json[i].name,
                "ename": json[i].ename,
                "departName": json[i].dep_name,
                "jobtitle": json[i].jobtitle,
                "eid": json[i].eid
            });
        }
        console.log(data);
        return data;
    }
